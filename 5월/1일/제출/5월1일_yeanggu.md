2.
SELECT EMPNO,ENAME,SAL 
  FROM emp 
 ORDER BY ENAME DESC;

3.
SELECT SAL,ENAME,EMPNO,LOC 
  FROM emp,dept 
 WHERE emp.DEPTNO = dept.DEPTNO 
 ORDER BY EMPNO ASC;

4.
SELECT ENAME,SAL*12 
  FROM emp 
 ORDER BY HIREDATE DESC;

5.
SELECT JOB 
  FROM emp 
 GROUP BY JOB;

6.
SELECT ENAME,MGR 
  FROM emp 
 WHERE EMPNO > 7521;

7.
SELECT SUM(SAL) 
  FROM emp 
 WHERE JOB = 'CLERK';

8.
SELECT GRADE,EMPNO,SAL 
  FROM emp,salgrade 
 WHERE SAL>LOSAL 
   AND SAL<HISAL 
   AND JOB = 'SALESMAN' 
 ORDER BY HIREDATE ASC;

9.
SELECT a.EMPNO,a.ENAME,b.ENAME 
  FROM emp AS a, 
  	   emp AS b 
 WHERE a.MGR = b.EMPNO;

10.
SELECT JOB,SUM(SAL) 
  FROM emp 
 GROUP BY JOB;

11.
SELECT JOB,SUM(SAL),RANK() OVER(ORDER BY SUM(SAL) DESC) 
  FROM emp 
 GROUP BY JOB;
